httpd:

nginx:
Mappear puerto: Docker run -d -p 5000:80 nginx

phpmyadmin:
Mappear puerto: Docker run -d -p 5000:80 phpmyadmin


CMABIAR NOMBRES DEL CONTENEDORES
HTTPD: docker run -d -p 5000:80 --name Apache httpd
NGINX: docker run -d -p 5001:80 --name Nginx nginx
PHPMYADMIN: docker run -d -p 5002:80 --name PHPAdmin
